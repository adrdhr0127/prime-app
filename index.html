const express = require('express'); const cors = require('cors'); const bcrypt = require('bcryptjs'); const jwt = require('jsonwebtoken'); const axios = require('axios'); const { PrismaClient } = require('@prisma/client'); require('dotenv').config(); const app = express(); const prisma = new PrismaClient(); app.use(cors()); app.use(express.json()); const JWT_SECRET = process.env.JWT_SECRET || 'fallback_secret_key'; // Middleware to check if user is logged in const authenticate = (req, res, next) => { const token = req.headers.authorization?.split(' ')[1]; if (!token) return res.status(401).json({ error: 'Unauthorized' }); try { req.user = jwt.verify(token, JWT_SECRET); next(); } catch (err) { res.status(403).json({ error: 'Invalid token' }); } }; // --- AUTHENTICATION --- app.post('/api/auth/register', async (req, res) => { try { const hashedPassword = await bcrypt.hash(req.body.password, 10); await prisma.user.create({ data: { username: req.body.username, password: hashedPassword } }); res.json({ message: 'Success' }); } catch (err) { res.status(400).json({ error: 'Username taken' }); } }); app.post('/api/auth/login', async (req, res) => { const user = await prisma.user.findUnique({ where: { username: req.body.username } }); if (!user || !(await bcrypt.compare(req.body.password, user.password))) { return res.status(401).json({ error: 'Invalid credentials' }); } const token = jwt.sign({ userId: user.id }, JWT_SECRET, { expiresIn: '24h' }); res.json({ token }); }); // --- PORTFOLIO & TRADING --- app.get('/api/portfolio', authenticate, async (req, res) => { const user = await prisma.user.findUnique({ where: { id: req.user.userId }, include: { positions: true } }); res.json(user); }); app.post('/api/trade', authenticate, async (req, res) => { const { symbol, type, shares, price } = req.body; const userId = req.user.userId; const totalCost = shares * price; try { await prisma.$transaction(async (tx) => { const user = await tx.user.findUnique({ where: { id: userId } }); if (type === 'BUY') { if (user.balance < totalCost) throw new Error("Insufficient funds"); await tx.user.update({ where: { id: userId }, data: { balance: user.balance - totalCost } }); const pos = await tx.position.findUnique({ where: { userId_symbol: { userId, symbol } } }); if (pos) { const newShares = pos.shares + shares; const newAvg = ((pos.shares * pos.avgPrice) + totalCost) / newShares; await tx.position.update({ where: { id: pos.id }, data: { shares: newShares, avgPrice: newAvg } }); } else { await tx.position.create({ data: { userId, symbol, shares, avgPrice: price } }); } } else if (type === 'SELL') { const pos = await tx.position.findUnique({ where: { userId_symbol: { userId, symbol } } }); if (!pos || pos.shares < shares) throw new Error("Insufficient shares"); await tx.user.update({ where: { id: userId }, data: { balance: user.balance + totalCost } }); if (pos.shares === shares) await tx.position.delete({ where: { id: pos.id } }); else await tx.position.update({ where: { id: pos.id }, data: { shares: pos.shares - shares } }); } await tx.transaction.create({ data: { userId, symbol, type, shares, price } }); }); res.json({ message: 'Trade successful' }); } catch (error) { res.status(400).json({ error: error.message }); } }); // --- API PROXIES --- app.get('/api/quote/:symbol', async (req, res) => { try { const response = await axios.get(`https://finnhub.io/api/v1/quote?symbol=${req.params.symbol}&token=${process.env.FINNHUB_API_KEY}`); res.json(response.data); } catch (err) { res.status(500).json({ error: 'Failed to fetch quote' }); } }); app.post('/api/analyze', authenticate, async (req, res) => { try { const prompt = `You are a financial AI. Analyze the stock ${req.body.symbol} currently priced at $${req.body.price}. Give a brief 3-sentence technical outlook.`; const response = await axios.post('https://api.anthropic.com/v1/messages', { model: "claude-3-haiku-20240307", max_tokens: 150, messages: [{ role: "user", content: prompt }] }, { headers: { 'x-api-key': process.env.ANTHROPIC_API_KEY, 'anthropic-version': '2023-06-01', 'content-type': 'application/json' } }); res.json({ analysis: response.data.content[0].text }); } catch (err) { res.status(500).json({ error: 'AI Analysis failed' }); } }); const PORT = process.env.PORT || 3000; app.listen(PORT, () => console.log(`Server running on port ${PORT}`));
